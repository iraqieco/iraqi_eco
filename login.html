document.getElementById("loginBtn").onclick = async () => { const email = document.getElementById("email").value.trim(); const password = document.getElementById("password").value; const msg = document.getElementById("msg"); msg.innerHTML = "جاري تسجيل الدخول..."; const { data, error } = await supabase.auth.signInWithPassword({ email, password }); if (error) { msg.innerHTML = error.message; alert(error.message); return; } alert("تم تسجيل الدخول"); location.href = "add.html"; };
